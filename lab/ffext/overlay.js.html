var tardis = {
  onLoad: function() {
    // initialization code
    this.initialized = true;
    this.strings = document.getElementById("tardis-strings");
  },
  onMenuItemCommand: function(e) {
    var pSvc = Components
      .classes["@mozilla.org/embedcomp/prompt-service;1"]
      .getService(Components.interfaces.nsIPromptService);
    pSvc.alert(window, this.strings.getString("whoMessageTitle"),
                       this.strings.getString("whoMessage"));
  },
};
window.addEventListener("load", function(e) {
  tardis.onLoad(e);
}, false);
